二分查找

2021/04/30

问题

https://leetcode-cn.com/problems/koko-eating-bananas/

code

class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        max_speed = max(piles)
        piles_len = len(piles)
        def can_eat(speed):
            tmp = map(lambda x: math.ceil(x/speed), piles)
            if sum(tmp) <= h: return True
            else: return False
        
        up, low = max_speed, 1
        while up > low:
            m = (low + up) // 2 
            if can_eat(m):
                up = m
            else: 
                low = m+1
            
        return low

如何确定 mid 向下取整还是向上取整
